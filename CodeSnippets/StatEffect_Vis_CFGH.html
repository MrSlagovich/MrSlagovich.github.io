// Stat effect type
UENUM(BlueprintType)
enum class EStatEffectType : uint8
{
    Active,
    Passive,
    Container,    // Used for having a base stat effect and another stat effect work at the same time
    VisualOnly, // Only adds a visual effect and doesn't effect gameplay
    OneShot        // Triggers once for one frame
};

/**
 *
 */
USTRUCT(BlueprintType)
struct LOOTCHADORS_API FGameplayParams
{
    GENERATED_USTRUCT_BODY()
    // Gameplay Tags to add to the player
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = StatEffect)
    FGameplayTagContainer PlayerStatTags;
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = StatEffect)
    EStatEffectType StatEffectType;
};

UCLASS(BlueprintType, Blueprintable)
class LOOTCHADORS_API UStatEffect_Gameplay_CFG : public UPrimaryDataAsset
{

    GENERATED_BODY()
public:
    // Gameplay parts of the CFG, Gameplay Tags to add and
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category = "StatEffect")
    FGameplayParams Gameplay;
};
