#include "StatEffect.h"


float AStatEffect::GetStatDuration_Implementation() {
    return StatDuration;
}

EStatEffectType AStatEffect::GetStatEffectType_Implementation()
{
    //if cfg is valid -> get from cfg
    if (StatEffectCFGs)
    {
        return StatEffectCFGs->cfgs.GameplayCFG->Gameplay.StatEffectType;
    }
    else
    {
        return EStatEffectType::Active;
    }
    
}
